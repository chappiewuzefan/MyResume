    {Australian National University} % Organisation
    {Binaural Signal Generation MCU Project} % Project
    {Canberra, Australia} % Location
    {May. 2023 - Jun. 2023} % Date(s)
    {
      \begin{cvitems} % Description(s) of project
        \item {Generated binaural signals from an audio file, utilizing a Microcontroller and MPU-6050}
        \item {Developed a system capable of measuring the azimuth angle with high precision, integrating it into the audio processing to create a spatial sound experience, and displaying the angle in real-time both on a connected LCD and through MATLAB visualization.}
        \item {Implemented data processing techniques like the "Overlap Save" method to handle large music files efficiently, overcoming the RAM limitation of the microcontroller and ensuring seamless audio generation.}
        \item {Designed the system to work with various communication protocols such as I2C, UART, and SPI, facilitating smooth integration and communication between multiple components including IMU, LCD, and SD-Card reader.}
        \item {Achieved efficient data communication and handling by optimizing the data transfer from music files to convolution, potentially leading to an increase in block size and a decrease in SD-Card access times.}
        \item {\textbf{Technical Skills:} Microcontroller Programming, Signal Processing, MATLAB, System Integration, I2C/UART/SPI Protocols, Real-time Data Visualization}
        \item {\textbf{Project Repository:} \url{https://github.com/chappiewuzefan/MCU_Project}}
      \end{cvitems}
    }

  \cventry
    {Australian National University} % Organisation
    {Smart Home Controller Project} % Project
    {Canberra, Australia} % Location
    {Mar 2023 - Apr 2023} % Date(s)
    {
      \begin{cvitems} % Description(s) of project
        \item {Designed and developed a smart home controller using a Basys3 FPGA board, integrating two vital functionalities: Secure Garage Door Control and Climate Control (Home Heating and Cooling), utilizing Verilog programming language.}
        \item {Implemented a secure garage door module featuring a finite state machine (FSM) with 14 distinct states, facilitating features such as pin-code security, alarm system, and collision detection, enhancing home security and automation.}
        \item {Developed a climate control module to regulate home heating and cooling, with an FSM to manage different operational stages based on user-defined and external temperature inputs, ensuring energy-efficient home climate management.}
        \item {Achieved seamless integration of both modules into a hierarchical structure, optimizing resource utilization and system management. Designed user interfaces with LED indicators and Seven-Segment Displays (SSDs) to provide real-time status and feedback.}
        \item {\textbf{Technical Skills:} Verilog, Digital Design, FPGA, Xilinx Vivado, System Integration, Finite State Machine (FSM)}
        \item {\textbf{Project Repository:} \url{https://github.com/chappiewuzefan/Smart-Home-Controller-Project}}
      \end{cvitems}
    }
